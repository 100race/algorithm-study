```python
class Solution(object):
    def isPalindrome(self, s):
        """
        :type s: str
        :rtype: bool
        """
        s = s.lower()
        result = False
        s_list = [x for x in s if x.isalnum()]
        s_len = len(s_list)
        
        if s_len % 2 == 0: #even
            if s_list[:s_len//2] == s_list[s_len//2 :s_len][::-1]: 
                result = True
            else:
                result = False
        else: #odd
            if s_list[:s_len//2] == s_list[s_len//2 +1 :s_len][::-1]:
                result = True
            else:
                result = False
        
        return result
```
